A requireJS module. For use with Game-Builder

This module serves as a container for all the viewport objects created.

This module extends delegate so it provides a few events to hook into:

ADD

When a viewport is added

Registered callbacks get the viewports as argument 

gb.viewports.on(gb.viewports.ADD, function(viewports) {});


REMOVE

When a viewport is removed

Registered callbacks get the viewport as argument 

gb.viewports.on(gb.viewports.REMOVE, function(viewport) {});


CHANGE

When a viewport changes position

Registered callbacks get the viewport as argument 

gb.viewports.on(gb.viewports.CHANGE, function(viewport) {});


MOVE

When all the viewport is moved

gb.viewports.on(gb.viewports.MOVE, function() {});


Note: The snippet uses the reference to viewports found in the gb module. The way you get a hold to a reference to viewports may vary.

Viewport management

define(["delegate", "viewport", "error-printer", "util"], function(Delegate, Viewport, ErrorPrinter, Util) { var viewports = {}; var viewportsArray = []; var ViewportContainer = Delegate.extend({ init: function () { this._super(); },

add

Creates a new viewport and adds it to the container.

Parameters:

  • name must be a String.
    (Id of the new viewport)

  • width must be a Number.
    (Width of the new viewport)

  • height must be a Number.
    (Height of the new viewport)

  • offsetX is optional, must be a Number, and has a default value of 0.
    (X offset relative to the top corner of the screen)

  • offsetY is optional, must be a Number, and has a default value of 0.
    (Y offset relative to the top corner of the screen)

  • scaleX is optional, must be a Number, and has a default value of 1.
    (X scale of the viewport)

  • scaleY is optional, must be a Number, and has a default value of 1.
    (Y scale of the viewport)

  • layers is optional, must be an Array, and has a default value of null.
    (Y An array of strings with the names of layer objects to add on the created viewport)

  • strokeColor is optional, must be a Number, and has a default value of null.
    (Color of the stroke around the viewports)

  • strokeWidth is optional, must be a Number, and has a default value of null.
    (Width of the stroke around the viewport )

  • worldfit is optional, must be a Boolean, and has a default value of false.
    (Whether viewport should scale to fit the world size)

  • culling is optional, must be a Boolean, and has a default value of true.
    (Whether the viewport should perform culling when game-objects are drawn to it)

  • clipping is optional, must be a Boolean, and has a default value of true.
    (Whether the viewport should clip drawings that fall outside of it's viewing area)

Returns an Object
(The newly created viewport. If it already exists, the existing one is returned)

add: function (name, width, height, offsetX, offsetY, scaleX, scaleY, layers, strokeColor, strokeWidth, worldFit, culling, clipping) { if (!viewports[name]) { var v = create.apply(this, arguments); viewports[name] = v; viewportsArray.push(v); this.execute(this.ADD, v); return v; } return viewports[name]; },

remove

Removes the specified viewport

Parameters:

  • name must be a String.
    (Id of the viewport to remove)
remove: function (name) { if (!viewports[name]) { ErrorPrinter.printError('Viewports', 'Viewport with id:' + name + ' does not exist.'); } var v = viewports[name]; viewportsArray.splice(viewportsArray.indexOf(v), 1); viewports[name] = null; delete viewports[name]; this.execute(this.REMOVE, v); v.destroy(); },

change

Change the position of the specified viewport

Parameters:

  • name is optional and must be a String.
    (Name viewport to change)

  • index is optional and must be a Number.
    (New index)

change: function (name, index) { var viewport = viewports[name]; var viewportIndex = viewportsArray.indexOf(viewport); viewportsArray.splice(viewportIndex, 1); viewportsArray.splice(index, 0, viewport); this.execute(this.CHANGE, viewport); },

after

Place the viewport specified by the first argument after the one specified by the second argument.

Parameters:

  • name is optional and must be a String.
    (Name viewport to change)

  • other is optional and must be a String.
    (Name viewport which should come before to the one in the first argument)

after: function (name, other) { move.call(this, 'after', name, other); },

before

Place the viewport specified by the first argument before the one specified by the second argument.

Parameters:

    (Name viewport to change)

  • other is optional and must be a String.
    (Name viewport which should come after to the one in the first argument)

before: function (name, other) { move.call(this, 'before', name, other); },

removeAll: function () { for (var i = viewportsArray.length - 1; i >= 0; i--) { this.remove(viewportsArray[i].name); } },

Iterate through all the viewports executing the callback function with each viewport as argument

Parameters:

iterate: function(scope, method) { for (var i = 0; i < viewportsArray.length; i++) { method.call(scope, viewportsArray[i]); } },

}); var create = function (name, width, height, offsetX, offsetY, scaleX, scaleY, layers, strokeColor, strokeWidth, worldFit, culling, clipping) { var v = new Viewport(name, width, height, offsetX, offsetY, scaleX, scaleY); v.setStroke(strokeWidth, strokeColor); if (Util.isBoolean(worldFit)) { v.WorldFit = worldFit; } if (Util.isBoolean(culling)) { v.Culling = culling; } if (Util.isBoolean(clipping)) { v.Clipping = clipping; } if (layers) { for (var i = 0; i < layers.length; i++) { v.addLayer(layers[i]); } } return v; } var move = function(type, current, pivot) { if (!viewports[current]) { ErrorPrinter.printError('Viewports', 'Viewport with id:' + current + ' does not exist.'); } if (!viewports[pivot]) { ErrorPrinter.printError('Viewports', 'Viewport with id:' + pivot + ' does not exist.'); } var current = viewports[current]; var pivot = viewports[pivot]; var currentIndex = viewportsArray.indexOf(current); var pivotIndex; if (type == 'before') { pivotIndex = viewportsArray.indexOf(pivot); } if (type == 'after') { pivotIndex = viewportsArray.indexOf(pivot) + 1; } viewportsArray.splice(pivotIndex, 0, viewportsArray.splice(currentIndex, 1)[0]); this.execute(this.MOVE, current); } Object.defineProperty(ViewportContainer.prototype, "ADD", { get: function() { return 'add'; } }); Object.defineProperty(ViewportContainer.prototype, "REMOVE", { get: function() { return 'remove'; } }); Object.defineProperty(ViewportContainer.prototype, "CHANGE", { get: function() { return 'change'; } }); Object.defineProperty(ViewportContainer.prototype, "MOVE", { get: function() { return 'move'; } }); return new ViewportContainer(); });
