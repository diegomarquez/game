A requireJS module. For use with Game-Builder

This is the type of objects that viewport uses to determine the order in which game-objects should be drawn. Each viewport has an array of this type of objects.

init: function(name, viewport) { this.name = name; this.gameObjects = []; this.visible = true; this.viewport = viewport; },

Add a game-object to layer for rendering

Parameters:

  • go must be an Object.
    (The game-object to add)

add: function(go) { var index = this.gameObjects.indexOf(go); if (index == -1) { this.gameObjects.push(go); go.addToViewportList(this.viewport.name, this.name); return go; } return null; },

remove

Remove a game-object from the layer

Parameters:

  • go must be an Object.
    (The game-object to remove)

Returns an Object or a null
(The game-object that was just removed or null if the game-object was not part of the layer)

remove: function(go) { var index = this.gameObjects.indexOf(go); if (index != -1) { this.gameObjects.splice(index, 1); go.removeFromViewportList(this.viewport.name, this.name); return go; } return null; },

removeAll

Removes all the game-objects from the layer

removeAll: function() { var gos = []; for (var i = 0; i < this.gameObjects.length; i++) { gos.push(this.remove(this.gameObjects[i])); } return gos; },
