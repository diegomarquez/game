A requireJS module. For use with Game-Builder

This is the type of objects that videport uses to determine the order in which game-objects should be drawn. Each viewport has an array of this type of objects.

Visual organization

define(["delegate"], function(Delegate){ var Layer = Delegate.extend({

init

Constructor

Parameters:

  • name must be a String.
    (The name of the layer)

  • viewport must be a Viewport.
    (The viewport the layer belongs to)

init: function(name, viewport) { this.name = name; this.gameObjects = []; this.visible = true; this.viewport = viewport; },

add

Add a game-obejct to layer for rendering

Parameters:

  • go must be an Object.
    (The game-object to add)
add: function(go) { this.gameObjects.push(go); },

remove

Remove a game-object from the layer

Parameters:

  • go must be an Object.
    (The game-object to remove)
remove: function(go) { this.gameObjects.splice(this.gameObjects.indexOf(go), 1); },

removeAll

Removes all the game-objects from the layer

removeAll: function() { this.gameObjects = []; },

draw

Parameters:

  • context must be a Context 2D.
    (Canvas 2D context)
draw: function(context) { if (!this.visible) return; var go; for (var i = 0; i < this.gameObjects.length; i++) { go = this.gameObjects[i]; if (this.viewport.isGameObjectInside(go, context)) { go.draw(context, this.viewport); } } },

show

Make the layer visible

show: function() { this.visible = true; },

hide

Make the layer invisible

hide: function() { this.visible = false; },

isVisible

Wether the layer is visible or not

Returns a Boolean

isVisible: function() { return this.visible; }
}); return Layer; });
