A requireJS module. For use with Game-Builder

Mainly this is a wrapper to the Image Object

This renderer is used to draw still images and can receive a bunch of configuration options when setting it up in the component-pool. ej.

gb.coPool.createConfiguration("Bitmap", 'Bitmap_Renderer')
     .args({ 
         //Path to the image you want to draw.
         //This is required
        path: 'some/path/to/image.jpg',

         //Use this if you want the registration point of the image to be the center
         //This is optional
        offset:'center',

        //If offset is not provided this two are used
        //These are optional and default to 0
        offsetX:0,
        offsetY:0, 

        //Use these to override the dimentions of the loaded image.
        //These are optional
        scaleX: 1, 
        scaleY: 1
});

Note: The snippet uses the reference to the component-pool found in the gb module. The way you get a hold to a reference to the component-pool may vary.

define(["renderer", "image-cache"], function(Renderer, ImageCache) { var image, w, h; var BitmapRenderer = Renderer.extend({

start

This is called by the game-object using this renderer. It sends the path configured to the image-cache module.

start: function(parent) { ImageCache.cache(this.path); },

draw

Draws the image into the canvas, applying configured properties, like width, height and offsets

Parameters:

  • context must be a Context 2D.
    (Canvas 2D context)
draw: function(context) { image = ImageCache.get(this.path); w = rendererWidth(); h = rendererHeight(); if (this.offset == 'center'){ context.drawImage(image, -w/2, -h/2, w, h); } else{ context.drawImage(image, this.rendererOffsetX(), this.rendererOffsetY(), w, h); } },

rendererWidth

Returns a Number
(The width of the renderer)

rendererWidth: function() { return ImageCache.get(this.path).width * this.scaleX; },

rendererHeight

Returns a Number
(The height of the renderer)

rendererHeight: function() { return ImageCache.get(this.path).height * this.scaleY; }
}); return BitmapRenderer; });
