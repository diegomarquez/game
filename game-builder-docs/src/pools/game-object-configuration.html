A requireJS module. For use with Game-Builder

This module describes the kind of objects that are held by the game-object-pool

Game Object Configuration

define(function(require) { var getComponentDescription = function(id, args) { return { componentId: id, args: args } } var GameObjectConfiguration = function(pool, type, alias) { this.type = type; this.alias = alias; this.hardArguments = null; this.childs = []; this.components = []; this.renderer = null; Object.defineProperty(this, "pool", { configurable: false, enumerable: false, writable: false, value: pool }); };

typeId

The id of the pool that objects are going to be taken off when building the game-object described in this configuration

Returns a String

GameObjectConfiguration.prototype.typeId = function() { return this.type; };

configurationId

The id the built game-object will get

Returns a String

GameObjectConfiguration.prototype.configurationId = function() { return this.alias; };

args

Set which arguments this configuration will apply to a game-object. This method is chainable

Parameters:

  • args must be an Object.

Returns an Object

GameObjectConfiguration.prototype.args = function(args) { this.hardArguments = args; this.pool.execute(this.pool.UPDATE_CONFIGURATION, this); return this; };
