A requireJS module. For use with Game-Builder

This module defines the object which is in charge of retrieving game-objects at any given point. It's the last piece of Game-Builder object management system.

It has various methods to send game-objects to their respective pools.

When you just want to get rid of everything, reclaimer has you covered.

Recycling

claim

Removes a game-object from it's parent if it has one, and then calls it's clear method. This sends it and all of the objects that depend of it to their respective pools.

Parameters:

  • go must be an Object.
    (game-object to recycle)

  • id must be a String.
    (Id assigned to the game-object in game-object-pool)

Can throw an Error
(If the id argument is missing.)

Reclaimer.prototype.claimWithId = function(go, id) { if(!go) { ErrorPrinter.missingArgumentError('Reclaimer', 'go'); } if(!id) { ErrorPrinter.missingArgumentError('Reclaimer', 'id'); } if(go.typeId == id || go.poolId == id) { this.claim(go); } }

claimType

Calls claimWithId on all the active game-objects that match the given type id.

Parameters:

  • typeName must be a String.
    (An id matching a existing type in game-object-pool)
Reclaimer.prototype.claimType = function(typeName) { var activeGameObjects = GameObjectPool.getActiveObjects(typeName); for (var i=activeGameObjects.length-1; i>=0; i--) { this.claimWithId(activeGameObjects[i], typeName); } };

Reclaimer.prototype.claimAllBut = function(mode, doNotClaim) { var allActiveGameObjects = GameObjectPool.getAllActiveObjects(); capitalizedMode = mode.charAt(0).toUpperCase() + mode.slice(1); for (var k in allActiveGameObjects) { var activeCollection = allActiveGameObjects[k]; for (var i=0; i<activeCollection.length; i++) { var go = activeCollection[i]; if(capitalizedMode == 'Configuration') { if (doNotClaim.indexOf(go.typeId) == -1) { this.claimWithId(go, go.typeId); } } if(capitalizedMode == 'Type') { if (doNotClaim.indexOf(go.poolId) == -1) { this.claimWithId(go, go.poolId); } } } } };
