A requireJS module. For use with Game-Builder

This module defines an extension that adds a bunch of group objects to the groups object which control the update of game-objects. The viewport objects are also defined here through the viewports object, these objects control the rendering of game-objects.

Also set here, are some shortcuts that are used when adding a game-object to a viewport layer.

To use a different layout, you can always define a new extension just like this one and replace the relevant bits.

Configure the world object

World.create(Gb.canvas.width*3, Gb.canvas.height*9);

Set a few update groups

Groups.add("First"); Groups.add("Second"); Groups.add("Third");

Create a viewport. It has a width and a height along with offset coordinates in case you don't want it to start in the top left corner of the canvas

var viewport = Viewports.add("Main", Gb.canvas.width-40, Gb.canvas.height-40, 20, 20);

Set the layers this viewport will have

viewport.addLayer("Back"); viewport.addLayer("Middle"); viewport.addLayer("Front"); viewport.setStroke(3, "#00FF00"); var mini = Viewports.add("Mini", Gb.canvas.width/3, Gb.canvas.height - 20, Gb.canvas.width/2 + 57, 10);

Scale the viewport so it can show all the world

World.scaleViewportToFit(mini); mini.addLayer("Front"); mini.setStroke(3, "#FF0000"); Gb.setViewportShortCut('MainBack', [{viewport:'Main', layer:'Back'}]); Gb.setViewportShortCut('MainMiddle', [{viewport:'Main', layer:'Middle'}]); Gb.setViewportShortCut('MainFront', [{viewport:'Main', layer:'Front'}]); Gb.setViewportShortCut('MainMiniFront', [{viewport:'Mini', layer:'Front'}, {viewport:'Main', layer:'Front'}]); Gb.setViewportShortCut('MiniFront', [{viewport:'Mini', layer:'Front'}]); }, destroy: function() { Groups.remove("First"); Groups.remove("Second"); Groups.remove("Third"); Viewports.remove('Main'); Viewports.remove('Mini'); Gb.removeViewportShortCut('MainBack'); Gb.removeViewportShortCut('MainMiddle'); Gb.removeViewportShortCut('MainFront'); Gb.removeViewportShortCut('MainMiniFront'); Gb.removeViewportShortCut('MiniFront'); } }); return BasicLayerSetup; });
